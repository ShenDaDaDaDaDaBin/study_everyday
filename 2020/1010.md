### for+while数组排序

```HTML
需求：
	对一个无序数组按从小到大的顺序排序

function sort (arr) {
	// 1、循环遍历数组
	for (let j = 1;j < arr.length;j++) {
        // 2、取数组第j项为当前项，赋值为常量
		const key = arr[j];
        let i = j - 1;
        // 3、取数组前一项，与当前项做对比
        while (i >= 0 && arr[i] > key) {
        	// 4、若前一项大于当前项，需把前一项的值赋给当前项，并对比j - 2 项与当前项值大小，若j- 2 项值也大于当前项，则需把j - 2 项的值赋给j - 1 项，从而实现大数的向后迁移
        	arr[i + 1] = arr[i];
        	i --;
        };
        // 5、当i小于0（即已经过数组第一项）或第i项小于当前项时，当前项赋给i下一项，即实现了当前及之前项的有序排列
        arr[i + 1] = key;
	}
    eg: [4,3,2]第一次for ——> [4,4,2]——> [3,4,2] 第二次for——> [3,3,4] ——> [2,3,4]
}
```
